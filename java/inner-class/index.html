Static Inner Class

class ClassA1{
  static class ClassB {
    void methodB() {
      System.out.println("methodB");
    }
  }
}

class ClassC {
  static class ClassD {
    void methodD() {
      System.out.println("methodD");
    }
  }
  public static void main(String[] args) {
    ClassA1.ClassB ab = new ClassA1.ClassB();
    ab.methodB();

    ClassD classD = new ClassD();
    classD.methodD();
  }
}
